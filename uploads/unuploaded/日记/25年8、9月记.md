import os
import subprocess

V2RAYN_PATH = r"D:\APP\v2rayN-windows-64-SelfContained\v2rayN.exe"  # v2rayN路径
GIT_BASH_PATH = r"C:\Program Files\Git\bin\bash.exe"  # Git Bash路径

def main():
    # 启动v2rayN
    subprocess.Popen(V2RAYN_PATH)  # 使用Popen而非run，无需等待v2rayN关闭
    
    # 转换Windows路径为Git Bash识别的路径格式
    bash_path = f"/{BLOG_FOLDER[0].lower()}/{BLOG_FOLDER[3:].replace('\\', '/')}"
    
    # 要执行的命令
    commands = f'cd "{bash_path}" && hexo g && hexo d && exit'
    
    # 启动Git Bash并执行命令
    subprocess.run([GIT_BASH_PATH, "-c", commands])
    
    input("按任意键退出...")

if __name__ == "__main__":
    main()

import subprocess
import time
import psutil
from pathlib import Path

# 配置参数
V2RAYN_PATH = r"D:\APP\v2rayN-windows-64-SelfContained\v2rayN.exe"  # v2rayN路径
GIT_BASH_PATH = r"C:\Program Files\Git\bin\bash.exe"  # Git Bash路径
BLOG_FOLDER = r"C:\BlogFile"  # 博客文件夹路径
V2RAYN_WAIT_SECONDS = 8  # v2rayN启动等待时间

def convert_windows_path_to_git_bash(windows_path):
    """将Windows路径转换为Git Bash可识别的路径格式"""
    print("正在转换Windows路径格式...")
    path_obj = Path(windows_path)
    drive_letter = path_obj.drive.replace(':', '').lower()
    relative_path = str(path_obj.relative_to(path_obj.anchor)).replace('\\', '/')
    result = f"/{drive_letter}/{relative_path}"
    print(f"路径转换完成: {windows_path} -> {result}")
    return result

def is_process_running(process_name):
    """检查进程是否正在运行"""
    for proc in psutil.process_iter(['name']):
        if proc.info['name'].lower() == process_name.lower():
            return True
    return False

def run_command_with_real_time_output(command, cwd=None):
    """运行命令并实时输出结果"""
    print(f"执行命令: {command}")
    process = None  # 初始化进程变量
    try:
        # 使用UTF-8编码处理输出
        process = subprocess.Popen(
            command,
            stdout=subprocess.PIPE,
            stderr=subprocess.STDOUT,
            universal_newlines=True,
            cwd=cwd,
            shell=True,
            encoding='utf-8',
            errors='replace'
        )
        
        # 实时输出命令执行结果
        while True:
            output = process.stdout.readline()
            if output == '' and process.poll() is not None:
                break
            if output:
                print(output.strip())
        
        return process.poll()
    except UnicodeDecodeError as e:
        print(f"UTF-8编码错误，尝试GBK编码: {e}")
        # 确保之前的进程已终止
        if process:
            try:
                process.terminate()
                time.sleep(1)
            except:
                pass
                
        # 尝试使用GBK编码
        process = subprocess.Popen(
            command,
            stdout=subprocess.PIPE,
            stderr=subprocess.STDOUT,
            universal_newlines=True,
            cwd=cwd,
            shell=True,
            encoding='gbk',
            errors='replace'
        )
        
        # 实时输出命令执行结果
        while True:
            output = process.stdout.readline()
            if output == '' and process.poll() is not None:
                break
            if output:
                print(output.strip())
        
        return process.poll()
    except Exception as e:
        print(f"命令执行错误: {e}")
        if process:
            try:
                process.terminate()
            except:
                pass
        return -1

def main():
    v2ray_process = None
    try:
        print("=" * 50)
        print("开始博客部署流程")
        print("=" * 50)
        
        # 步骤1: 启动v2rayN
        print("\n1. 正在启动v2rayN...")
        
        # 检查v2rayN是否已在运行
        v2ray_name = os.path.basename(V2RAYN_PATH)
        if is_process_running(v2ray_name):
            print(f"⚠️  {v2ray_name}已在运行，无需重复启动")
            v2ray_process = None  # 标记为未启动新进程
        else:
            v2ray_process = subprocess.Popen(V2RAYN_PATH)
            print("✅ v2rayN启动命令已发送")
        
        # 等待v2rayN完全启动
        print(f"等待v2rayN启动完成 ({V2RAYN_WAIT_SECONDS}秒)...")
        for i in range(V2RAYN_WAIT_SECONDS, 0, -1):
            print(f"倒计时: {i}秒")
            time.sleep(1)
        
        # 步骤2: 转换路径格式
        print("\n2. 正在转换路径格式...")
        bash_path = convert_windows_path_to_git_bash(BLOG_FOLDER)
        
        # 步骤3: 执行Hexo命令
        print("\n3. 正在执行Hexo部署命令...")
        
        # 构建通过Git Bash执行的命令前缀
        git_bash_command_prefix = f'"{GIT_BASH_PATH}" -c "cd {bash_path} && '
        
        # 执行Hexo清理
        print("\n执行: hexo clean")
        clean_command = f'{git_bash_command_prefix} hexo clean"'
        return_code = run_command_with_real_time_output(clean_command)
        if return_code != 0:
            print("❌ Hexo清理失败!")
            return
        
        print("✅ Hexo清理完成")
        
        # 执行Hexo生成
        print("\n执行: hexo generate")
        generate_command = f'{git_bash_command_prefix} hexo generate"'
        return_code = run_command_with_real_time_output(generate_command)
        if return_code != 0:
            print("❌ Hexo生成失败!")
            return
            
        print("✅ Hexo生成完成")
        
        # 执行Hexo部署
        print("\n执行: hexo deploy")
        deploy_command = f'{git_bash_command_prefix} hexo deploy"'
        return_code = run_command_with_real_time_output(deploy_command)
        if return_code != 0:
            print("❌ Hexo部署失败!")
            return
            
        print("✅ Hexo部署完成")
        
        print("\n" + "=" * 50)
        print("🎉 博客部署成功完成!")
        print("=" * 50)
            
    except Exception as e:
        print(f"\n❌ 发生错误: {e}")
        import traceback
        traceback.print_exc()
    finally:
        # 移除关闭v2rayN的逻辑，保持其运行状态
        print("\n清理完成")
        if v2ray_process:
            print(f"ℹ️  v2rayN将继续运行，无需特殊处理")
        else:
            print(f"ℹ️  v2rayN状态保持不变")
        
        print("\n提示：若需关闭v2rayN，可在系统托盘找到对应图标手动关闭")
            
    input("\n按Enter键退出...")

if __name__ == "__main__":
    main()
